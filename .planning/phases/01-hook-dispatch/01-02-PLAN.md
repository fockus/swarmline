2. Modify `execute()` method:
   ```python
   async def execute(self, tool_name: str, args: dict[str, Any]) -> str:
       # PreToolUse hook dispatch
       effective_args = args
       if self._hook_dispatcher is not None:
           pre_result = await self._hook_dispatcher.dispatch_pre_tool(tool_name, args)
           if pre_result.action == "block":
               return json.dumps(
                   {"error": f"Tool '{tool_name}' blocked by hook: {pre_result.message}"},
                   ensure_ascii=False,
               )
           if pre_result.action == "modify" and pre_result.modified_input is not None:
               effective_args = pre_result.modified_input
       
       # Execute tool (use effective_args instead of args)
       if tool_name in self._local_tools:
           result = await self._execute_local(tool_name, effective_args)
       elif tool_name.startswith("mcp__"):
           result = await self._execute_mcp(tool_name, effective_args)
       else:
           return json.dumps({"error": f"Tool '{tool_name}' not found"}, ensure_ascii=False)
       
       # PostToolUse hook dispatch
       if self._hook_dispatcher is not None:
           modified = await self._hook_dispatcher.dispatch_post_tool(tool_name, effective_args, result)
           if modified is not None:
               result = modified
       
       return result
   ```

3. Update `_execute_local` and `_execute_mcp` to accept args parameter (they already do).

Import: `from swarmline.hooks.dispatcher import HookDispatcher` (lazy import inside TYPE_CHECKING or at top).
Use TYPE_CHECKING to avoid circular imports if needed.
